00025 void* PoolThread::run(void *arg)
00026 {
00027         while (true)
00028         {
00029                 PoolThread* ths = (PoolThread*)arg;
00030                 Thread* mthread = ths->mthread;
00031                 //cout << "started wait" << endl;
00032                 mthread->wait();
00033                 //cout << "end wait" << endl;
00034                 ths = (PoolThread*)arg;
00035                 Task* task = ths->task;
00036                 bool console = ths->console;
00037                 if (task != NULL)
00038                 {
00039                         if (task->tunit != -1 && task->type != -1 && task->tunit > 0 && task->type > 0)
00040                         {
00041                                 if(console)
00042                                 {
00043                                         ths->logger << "Task scheduled for sleep\n" << flush;
00044                                 }
00045                                 if (task->type == TimeUnit::NANOSECONDS)
00046                                         Thread::nSleep(task->tunit);
00047                                 else if (task->type == TimeUnit::MICROSECONDS)
00048                                         Thread::uSleep(task->tunit);
00049                                 else if (task->type == TimeUnit::MILLISECONDS)
00050                                         Thread::mSleep(task->tunit);
00051                                 else if (task->type == TimeUnit::SECONDS)
00052                                         Thread::sSleep(task->tunit);
00053                                 else if (task->type == TimeUnit::MINUTES)
00054                                         Thread::nSleep(task->tunit * 60);
00055                                 else if (task->type == TimeUnit::HOURS)
00056                                         Thread::nSleep(task->tunit * 60 * 60);
00057                                 else if(task->type==TimeUnit::DAYS)
00058                                         Thread::nSleep(task->tunit * 24 * 60 * 60);
00059                         }
00060                         try
00061                         {
00062                                 task->run();
00063                         }
00064                         catch(exception& e)
00065                         {
00066                                 if(console)
00067                                 {
00068                                         ths->logger << e.what() << flush;
00069                                 }
00070                         }
00071                         task->~Task();
00072                         ths->task = NULL;
00073                 }
00074                 ths->idle = true;
00075         }
00076         return NULL;
00077 }
