00068 SSL_CTX *SSLHandler::initialize_ctx(char *keyfile,char *password, string ca_list)
00069   {
00070     SSL_METHOD *meth;
00071     SSL_CTX *ctx;
00072 
00073     if(!bio_err){
00074       /* Global system initialization*/
00075       SSL_library_init();
00076       SSL_load_error_strings();
00077 
00078       /* An error write context */
00079       bio_err=BIO_new_fp(stderr,BIO_NOCLOSE);
00080     }
00081 
00082     /* Set up a SIGPIPE handler */
00083     signal(SIGPIPE,sigpipe_handle);
00084 
00085     /* Create our context*/
00086     meth=(SSL_METHOD*)SSLv23_method();
00087     ctx=SSL_CTX_new(meth);
00088 
00089     /* Load our keys and certificates*/
00090     if(!(SSL_CTX_use_certificate_chain_file(ctx,
00091       keyfile)))
00092         logger << "Can't read certificate file" << flush;
00093 
00094     pass=password;
00095     SSL_CTX_set_default_passwd_cb(ctx,
00096       password_cb);
00097     if(!(SSL_CTX_use_PrivateKey_file(ctx,
00098       keyfile,SSL_FILETYPE_PEM)))
00099         logger << "Can't read key file" << flush;
00100 
00101     /* Load the CAs we trust*/
00102     if(!(SSL_CTX_load_verify_locations(ctx,
00103       ca_list.c_str(),0)))
00104         logger << "Can't read CA list" << flush;
00105 #if (OPENSSL_VERSION_NUMBER < 0x00905100L)
00106     SSL_CTX_set_verify_depth(ctx,1);
00107 #endif
00108 
00109     return ctx;
00110   }

00117 void SSLHandler::error_occurred(char *error,int fd,SSL *ssl)
00118 {
00119         logger << error << flush;
00120         close(fd);
00121         int r=SSL_shutdown(ssl);
00122         if(!r){
00123           /* If we called SSL_shutdown() first then
00124                  we always get return value of '0'. In
00125                  this case, try again, but first send a
00126                  TCP FIN to trigger the other side's
00127                  close_notify*/
00128           shutdown(fd,1);
00129           r=SSL_shutdown(ssl);
00130         }
00131         switch(r){
00132           case 1:
00133                 break; /* Success */
00134           case 0:
00135           case -1:
00136           default:
00137                   logger << "shutdown failed" << flush;
00138                   break;
00139         }
00140         SSL_free(ssl);
00141 }
