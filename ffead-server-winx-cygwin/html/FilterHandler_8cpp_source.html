00034 void FilterHandler::handleIn(HttpRequest* req, HttpResponse& res, map<string, vector<string> > filterMap, void* dlib,
00035                 string ext)
00036 {
00037         Logger logger = Logger::getLogger("FilterHandler");
00038         if(filterMap.find(req->getCntxt_name()+"*.*in")!=filterMap.end() || filterMap.find(req->getCntxt_name()+ext+"in")!=filterMap.end())
00039         {
00040                 vector<string> tempp;
00041                 if(filterMap.find(req->getCntxt_name()+"*.*in")!=filterMap.end())
00042                         tempp = filterMap[req->getCntxt_name()+"*.*in"];
00043                 else
00044                         tempp = filterMap[req->getCntxt_name()+ext+"in"];
00045 
00046                 for (int var = 0; var < (int)tempp.size(); ++var)
00047                 {
00048                         string clasz = tempp.at(var);
00049                         clasz = "getReflectionCIFor" + clasz;
00050                         logger << "filter handled by class " << clasz << " " << dlib << endl;
00051                         if(dlib == NULL)
00052                         {
00053                                 logger << "error" << endl;
00054                                 cerr << dlerror() << endl;
00055                                 exit(-1);
00056                         }
00057                         void *mkr = dlsym(dlib, clasz.c_str());
00058                         if(mkr!=NULL)
00059                         {
00060                                 FunPtr f =  (FunPtr)mkr;
00061                                 ClassInfo srv = f();
00062                                 args argus;
00063                                 Constructor ctor = srv.getConstructor(argus);
00064                                 Reflector ref;
00065                                 void *_temp = ref.newInstanceGVP(ctor);
00066                                 Filter *filter = (Filter*)_temp;
00067                                 filter->doInputFilter(req);
00068                                 logger << "filter called" << endl;
00069                                 delete _temp;
00070                         }
00071                 }
00072         }
