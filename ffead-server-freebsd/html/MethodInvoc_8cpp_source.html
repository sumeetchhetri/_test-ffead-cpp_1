00048         methInfo =methInfo.substr(0,methInfo.find_last_of(">")+1);
00049         try
00050         {
00051                 XmlParser parser("Parser");
00052                 if(methInfo.find("lang=\"c++\"")!=string::npos || methInfo.find("lang='c++'")!=string::npos)
00053                 {
00054                         Document doc = parser.getDocument(methInfo);
00055                         Element message = doc.getRootElement();
00056                         if(message.getTagName()!="method")
00057                         {
00058                                 throw new MethodInvokerException("No method Tag\n",retValue);
00059                         }
00060                         if(message.getAttributes().size()<3)
00061                         {
00062                                 throw new MethodInvokerException("name,class and lang are mandatory attributes\n",retValue);
00063                         }
00064                         else if(message.getAttribute("name")=="")
00065                         {
00066                                 throw new MethodInvokerException("name attribute missing\n",retValue);
00067                         }
00068                         else if(message.getAttribute("className")=="")
00069                         {
00070                                 throw new MethodInvokerException("class attribute missing\n",retValue);
00071                         }
00072                         else if(message.getAttribute("lang")=="")
00073                         {
00074                                 throw new MethodInvokerException("lang attribute missing\n",retValue);
00075                         }
00076                         if(message.getChildElements().size()!=1)
00077                         {
00078                                 throw new MethodInvokerException("message tag should have only one child tag\n",retValue);
00079                         }
00080                         else if(message.getChildElements().at(0).getTagName()!="args")
00081                         {
00082                                 throw new MethodInvokerException("message tag should have an args child tag\n",retValue);
00083                         }
00084                         Serialize ser;
00085                         Reflector reflector;
00086                         args argus;
00087                         vals valus;
00088                         ElementList argts = message.getChildElements().at(0).getChildElements();
00089                         for (unsigned var = 0; var < argts.size();  var++)
00090                         {
00091                                 void *value = NULL;
00092                                 Element arg = argts.at(var);
00093                                 if(arg.getTagName()!="argument" || arg.getAttribute("type")=="")
00094                                         throw new MethodInvokerException("every argument tag should have a name and type attribute\n",retValue);
00095                                 if(arg.getText()=="" && arg.getChildElements().size()==0)
00096                                         throw new MethodInvokerException("argument value missing\n",retValue);
00097                                 if(arg.getAttribute("type")=="int")
00098                                 {
00099                                         int *vt = new int;
00100                                         *vt = boost::lexical_cast<int>(arg.getText());
00101                                         value = vt;
00102                                 }
00103                                 else if(arg.getAttribute("type")=="float")
00104                                 {
00105                                         float *vt = new float;
00106                                         *vt = boost::lexical_cast<float>(arg.getText());
00107                                         value = vt;
00108                                 }
00109                                 else if(arg.getAttribute("type")=="double")
00110                                 {
00111                                         double *vt = new double;
00112                                         *vt = boost::lexical_cast<double>(arg.getText());
00113                                         value = vt;
00114                                 }
00115                                 else if(arg.getAttribute("type")=="string")
00116                                 {
00117                                         string *vt = new string;
00118                                         *vt = boost::lexical_cast<string>(arg.getText());
00119                                         value = vt;
00120                                 }
00121                                 else if(arg.getAttribute("type")!="")
00122                                 {
00123                                         Element obj = arg.getChildElements().at(0);
00124                                         string objxml = obj.render();
00125                                         string objClassName = obj.getTagName();
00126                                         value = ser.unSerializeUnknown(objxml,arg.getAttribute("type"));
00127                                 }
00128                                 argus.push_back(arg.getAttribute("type"));
00129                                 valus.push_back(value);
00130                         }
00131                         string className = message.getAttribute("className");
00132                         string returnType = message.getAttribute("returnType");
00133                         string lang = message.getAttribute("lang");
00134                         ClassInfo clas = reflector.getClassInfo(className);
00135                         string methodName = message.getAttribute("name");;
00136                         if(clas.getClassName()=="")
00137                         {
00138                                 throw new MethodInvokerException("class does not exist or is not in the library path\n",retValue);
00139                         }
00140                         Method meth = clas.getMethod(methodName,argus);
00141                         if(meth.getMethodName()=="")
00142                         {
00143                                 throw new MethodInvokerException("method does not exist for the class or the class does not exist in the library path\n",retValue);
00144                         }
00145                         else
00146                         {
00147                                 args argus;
00148                                 Constructor ctor = clas.getConstructor(argus);
00149                                 void *_temp = reflector.newInstanceGVP(ctor);
00150                                 if(returnType=="void" || returnType=="")
00151                                 {
00152                                         reflector.invokeMethod<void*>(_temp,meth,valus);
00153                                         retValue = ("<return:void></return:void>");
00154                                 }
00155                                 else
00156                                 {
00157                                         if(returnType=="int")
00158                                         {
00159                                                 int retv = reflector.invokeMethod<int>(_temp,meth,valus);
00160                                                 retValue = ("<return:int>"+boost::lexical_cast<string>(retv)+"</return:int>");
00161                                         }
00162                                         else if(returnType=="float")
00163                                         {
00164                                                 float retv = reflector.invokeMethod<float>(_temp,meth,valus);
00165                                                 retValue = ("<return:float>"+boost::lexical_cast<string>(retv)+"</return:float>");
00166                                         }
00167                                         else if(returnType=="double")
00168                                         {
00169                                                 double retv = reflector.invokeMethod<double>(_temp,meth,valus);
00170                                                 retValue = ("<return:double>"+boost::lexical_cast<string>(retv)+"</return:double>");
00171                                         }
00172                                         else if(returnType=="string")
00173                                         {
00174                                                 string retv = reflector.invokeMethod<string>(_temp,meth,valus);
00175                                                 retValue = ("<return:string>"+retv+"</return:string>");
00176                                         }
00177                                         else if(returnType!="")
00178                                         {
00179                                                 void* retobj = reflector.invokeMethodUnknownReturn(_temp,meth,valus);
00180                                                 string oxml = ser.serializeUnknown(retobj,returnType);
00181                                                 retValue = ("<return:"+returnType+">"+oxml+"</return:"+returnType+">");
00182                                         }
00183                                 }
00184 
00185                         }
00186                 }
00187                 else
00188                 {
00189                         retValue = "<return:exception>This is a C++ daemon</return:exception>";
00190                 }
00191                 if(retValue!="")
00192                         _methinv_instance->getServer()->Send(fd,retValue);
00193                 close(fd);
00194         }
00195         catch(MethodInvokerException *e)
00196         {
00197                 _methinv_instance->getServer()->Send(fd,retValue);
00198                 close(fd);
00199         }
00200         catch(...)
00201         {
00202                 retValue = ("<return:exception>XmlParseException occurred</return:exception>");
00203                 _methinv_instance->getServer()->Send(fd,retValue);
00204                 close(fd);
