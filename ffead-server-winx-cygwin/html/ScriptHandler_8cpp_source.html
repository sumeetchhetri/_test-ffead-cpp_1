00035 int ScriptHandler::popenRWE(int *rwepipe, const char *exe, const char *const argv[],string tmpf)
00036 {
00037         int in[2];
00038         int out[2];
00039         int err[2];
00040         int pid;
00041         int rc;
00042 
00043         rc = pipe(in);
00044         if (rc<0)
00045                 goto error_in;
00046 
00047         rc = pipe(out);
00048         if (rc<0)
00049                 goto error_out;
00050 
00051         rc = pipe(err);
00052         if (rc<0)
00053                 goto error_err;
00054 
00055         pid = fork();
00056         if (pid > 0) { // parent
00057                 close(in[0]);
00058                 close(out[1]);
00059                 close(err[1]);
00060                 rwepipe[0] = in[1];
00061                 rwepipe[1] = out[0];
00062                 rwepipe[2] = err[0];
00063                 return pid;
00064         } else if (pid == 0) { // child
00065                 close(in[1]);
00066                 close(out[0]);
00067                 close(err[0]);
00068                 close(0);
00069                 dup(in[0]);
00070                 close(1);
00071                 dup(out[1]);
00072                 close(2);
00073                 dup(err[1]);
00074                 logger << tmpf << endl;
00075                 chdir(tmpf.c_str());
00076                 execvp(exe, (char**)argv);
00077                 exit(1);
00078         } else
00079                 goto error_fork;
00080 
00081         return pid;
00082 
00083 error_fork:
00084         close(err[0]);
00085         close(err[1]);
00086 error_err:
00087         close(out[0]);
00088         close(out[1]);
00089 error_out:
00090         close(in[0]);
00091         close(in[1]);
00092 error_in:
00093         return -1;
00094 }
