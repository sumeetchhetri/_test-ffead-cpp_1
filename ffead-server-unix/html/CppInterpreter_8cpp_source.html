00338 bool CppInterpreter::evaluateCondition(string condition)
00339 {
00340         bool state = false;
00341         string token;
00342         if(condition.find("==")!=string::npos)
00343         {
00344                 token = "==";
00345         }
00346         else if(condition.find("!=")!=string::npos)
00347         {
00348                 token = "!=";
00349         }
00350         else if(condition.find("<=")!=string::npos || condition.find("=<")!=string::npos)
00351         {
00352                 token = "<=";
00353         }
00354         else if(condition.find(">=")!=string::npos || condition.find("=>")!=string::npos)
00355         {
00356                 token = ">=";
00357         }
00358         else if(condition.find(">")!=string::npos)
00359         {
00360                 token = ">";
00361         }
00362         else if(condition.find("<")!=string::npos)
00363         {
00364                 token = "<";
00365         }
00366         if(token!="")
00367         {
00368                 vector<string> bs;
00369                 StringUtil::split(bs, condition, (token));
00370                 if(bs.size()==2 && bs.at(0)!="" && bs.at(1)!="")
00371                 {
00372                         Object source,target;
00373                         if(localVariables.find(bs.at(0))!=localVariables.end())
00374                         {
00375                                 source = localVariables[bs.at(0)];
00376                         }
00377                         else if(boundVariables.find(bs.at(0))!=boundVariables.end())
00378                         {
00379                                 source = boundVariables[bs.at(0)];
00380                         }
00381                         if(containsChar(bs.at(1)))
00382                         {
00383                                 if(localVariables.find(bs.at(1))!=localVariables.end())
00384                                 {
00385                                         target = localVariables[bs.at(1)];
00386                                 }
00387                                 else if(boundVariables.find(bs.at(1))!=boundVariables.end())
00388                                 {
00389                                         target = boundVariables[bs.at(1)];
00390                                 }
00391                                 state = retState(token,source,target);
00392                         }
00393                         else
00394                         {
00395                                 state = retState(token,source,bs.at(1));
00396                         }
00397                 }
00398         }
00399         return state;
