00067 bool MessageUtil::sendMessage(Message msg)
00068 {
00069         int sockfd, numbytes;
00070         char buf[100];
00071         struct addrinfo hints, *servinfo, *p;
00072         int rv;
00073         char s[INET6_ADDRSTRLEN];
00074 
00075         memset(&hints, 0, sizeof hints);
00076         hints.ai_family = AF_UNSPEC;
00077         hints.ai_socktype = SOCK_STREAM;
00078 
00079         if ((rv = getaddrinfo(this->host.c_str(), this->port.c_str(), &hints, &servinfo)) != 0) {
00080                 fprintf(stderr, "getaddrinfo: %s\n", gai_strerror(rv));
00081                 return 1;
00082         }
00083 
00084         // loop through all the results and connect to the first we can
00085         for(p = servinfo; p != NULL; p = p->ai_next) {
00086                 if ((sockfd = socket(p->ai_family, p->ai_socktype,
00087                                 p->ai_protocol)) == -1) {
00088                         perror("client: socket");
00089                         continue;
00090                 }
00091 
00092                 if (connect(sockfd, p->ai_addr, p->ai_addrlen) == -1) {
00093                         close(sockfd);
00094                         perror("client: connect");
00095                         continue;
00096                 }
00097 
00098                 break;
00099         }
00100 
00101         if (p == NULL) {
00102                 fprintf(stderr, "client: failed to connect\n");
00103                 return 2;
00104         }
00105 
00106         inet_ntop(p->ai_family, get_in_addr((struct sockaddr *)p->ai_addr),
00107                         s, sizeof s);
00108         printf("client: connecting to %s\n", s);
00109 
00110         freeaddrinfo(servinfo); // all done with this structure
00111 
00112         string h = msg.toXml();
00113         bool flag = true;
00114         if (send(sockfd, h.c_str(), h.length(), 0) == -1)
00115         {
00116                 logger << "send failed" << flush;
00117                 flag = false;
00118         }
00119         logger << h << flush;
00120 
00121         if ((numbytes = recv(sockfd, buf, 99, 0)) == -1) {
00122                 perror("recv");
00123                 exit(1);
00124         }
00125 
00126         buf[numbytes] = '\0';
00127         printf("client: received '%s'\n",buf);
00128         close(sockfd);
00129         memset(&buf[0], 0, sizeof(buf));
00130         return flag;
