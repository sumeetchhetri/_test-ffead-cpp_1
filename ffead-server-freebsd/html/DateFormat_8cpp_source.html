00040 string DateFormat::format(Date date)
00041 {
00042         string temp = this->formatspec;
00043         StringUtil::replaceAll(temp,"hh",date.getHhStr());
00044         StringUtil::replaceAll(temp,"mi",date.getMmStr());
00045         StringUtil::replaceAll(temp,"ss",date.getSsStr());
00046         StringUtil::replaceAll(temp,"ns",date.getNsStr());
00047         StringUtil::replaceAll(temp,"ddd",date.getDayw());
00048         StringUtil::replaceAll(temp,"dd",date.getDayStr());
00049         StringUtil::replaceAll(temp,"mmm",date.getMonthw());
00050         StringUtil::replaceAll(temp,"mm",date.getMonthStr());
00051         StringUtil::replaceAll(temp,"yyyy",date.getYearStr());
00052         StringUtil::replaceAll(temp,"yy",date.getYearStr().substr(2));
00053         return temp;
00054 }
