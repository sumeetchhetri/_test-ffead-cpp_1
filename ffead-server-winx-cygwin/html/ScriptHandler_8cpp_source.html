00041 string ScriptHandler::execute(string exe, bool retErrs)
00042 {
00043         string data;
00044         if(retErrs)
00045         {
00046                 exe += " 2>&1";
00047         }
00048         FILE *pp;
00049         pp = popen(exe.c_str(), "r");
00050         if (pp != NULL) {
00051                 while (1) {
00052                   char *line;
00053                   char buf[1000];
00054                   line = fgets(buf, sizeof buf, pp);
00055                   if (line == NULL) break;
00056                   data.append(line);
00057                   //if (line[0] == 'd') printf("%s", line); /* line includes '\n' */
00058                 }
00059                 pclose(pp);
00060         }
00061         return data;
00062 }

00064 int ScriptHandler::popenRWE(int *rwepipe, const char *exe, const char *const argv[],string tmpf)
00065 {
00066         int in[2];
00067         int out[2];
00068         int err[2];
00069         int pid;
00070         int rc;
00071 
00072         rc = pipe(in);
00073         if (rc<0)
00074                 goto error_in;
00075 
00076         rc = pipe(out);
00077         if (rc<0)
00078                 goto error_out;
00079 
00080         rc = pipe(err);
00081         if (rc<0)
00082                 goto error_err;
00083 
00084         pid = fork();
00085         if (pid > 0) { // parent
00086                 close(in[0]);
00087                 close(out[1]);
00088                 close(err[1]);
00089                 rwepipe[0] = in[1];
00090                 rwepipe[1] = out[0];
00091                 rwepipe[2] = err[0];
00092                 return pid;
00093         } else if (pid == 0) { // child
00094                 close(in[1]);
00095                 close(out[0]);
00096                 close(err[0]);
00097                 close(0);
00098                 dup(in[0]);
00099                 close(1);
00100                 dup(out[1]);
00101                 close(2);
00102                 dup(err[1]);
00103                 logger << tmpf << endl;
00104                 chdir(tmpf.c_str());
00105                 execvp(exe, (char**)argv);
00106                 exit(1);
00107         } else
00108                 goto error_fork;
00109 
00110         return pid;
00111 
00112 error_fork:
00113         close(err[0]);
00114         close(err[1]);
00115 error_err:
00116         close(out[0]);
00117         close(out[1]);
00118 error_out:
00119         close(in[0]);
00120         close(in[1]);
00121 error_in:
00122         return -1;
00123 }
