00033 void Trace::trace()
00034 {
00035         using namespace abi;
00036         enum{MAX_DEPTH = 10};
00037         void *trace[MAX_DEPTH];
00038         Dl_info dlinfo;
00039         int trace_size = backtrace(trace, MAX_DEPTH);
00040         int status;
00041         const char *symname;
00042         char *demangled;
00043         logger << "Call stack: " << endl;
00044 
00045         for (int i=0; i<trace_size; ++i)
00046         {
00047                 if(!dladdr(trace[i], &dlinfo))
00048                         continue;
00049 
00050                 symname = dlinfo.dli_sname;
00051                 demangled = __cxa_demangle(symname, NULL, 0, &status);
00052                 if(status == 0 && demangled)
00053                         symname = demangled;
00054                 logger << "object: " << dlinfo.dli_fname << ", function: " << symname << endl;
00055                 if (demangled)
00056                         free(demangled);
00057         }
00058 }
