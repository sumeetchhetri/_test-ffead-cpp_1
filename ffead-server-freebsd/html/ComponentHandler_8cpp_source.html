00046 void* ComponentHandler::service(void* arg)
00047 {
00048         int fd = *(int*)arg;
00049         init();
00050         string methInfo,retValue;
00051         _cmp_instance->getServer()->Receive(fd,methInfo,1024);
00052         methInfo =methInfo.substr(0,methInfo.find_last_of(">")+1);
00053         _cmp_instance->logger << methInfo << flush;
00054         try
00055         {
00056                 XmlParser parser("Parser");
00057                 _cmp_instance->logger << "\nBean call parsed successfully\n" << flush;
00058                 if(methInfo.find("lang=\"c++\"")!=string::npos || methInfo.find("lang='c++'")!=string::npos)
00059                 {
00060                         Document doc = parser.getDocument(methInfo);
00061                         Element message = doc.getRootElement();
00062                         if(message.getTagName()!="service")
00063                         {
00064                                 throw new ComponentHandlerException("No service Tag\n",retValue);
00065                         }
00066                         if(message.getAttributes().size()<4)
00067                         {
00068                                 throw new ComponentHandlerException("name,beanName,returnType and lang are mandatory attributes\n",retValue);
00069                         }
00070                         else if(message.getAttribute("name")=="")
00071                         {
00072                                 throw new ComponentHandlerException("name attribute missing\n",retValue);
00073                         }
00074                         else if(message.getAttribute("returnType")=="")
00075                         {
00076                                 throw new ComponentHandlerException("returnType attribute missing\n",retValue);
00077                         }
00078                         else if(message.getAttribute("beanName")=="")
00079                         {
00080                                 throw new ComponentHandlerException("beanName attribute missing\n",retValue);
00081                         }
00082                         else if(message.getAttribute("lang")=="")
00083                         {
00084                                 throw new ComponentHandlerException("lang attribute missing\n",retValue);
00085                         }
00086                         if(message.getChildElements().size()!=1)
00087                         {
00088                                 throw new ComponentHandlerException("message tag should have only one child tag\n",retValue);
00089                         }
00090                         else if(message.getChildElements().at(0).getTagName()!="args")
00091                         {
00092                                 throw new ComponentHandlerException("message tag should have an args child tag\n",retValue);
00093                         }
00094                         Serialize ser;
00095                         Reflector reflector;
00096                         args argus;
00097                         vals valus;
00098                         ElementList argts = message.getChildElements().at(0).getChildElements();
00099                         for (unsigned var = 0; var < argts.size();  var++)
00100                         {
00101                                 void *value = NULL;
00102                                 Element arg = argts.at(var);
00103                                 if(arg.getTagName()!="argument")
00104                                         throw new ComponentHandlerException("Invalid Tag, only argument tag allowed\n",retValue);
00105                                 else if(arg.getAttribute("type")=="")
00106                                         throw new ComponentHandlerException("every argument tag should have a type attribute\n",retValue);
00107                                 if(arg.getText()=="" && arg.getChildElements().size()==0)
00108                                         throw new ComponentHandlerException("argument value missing\n",retValue);
00109                                 if(arg.getAttribute("type")!="")
00110                                 {
00111                                         Element obj = arg.getChildElements().at(0);
00112                                         string objxml = obj.render();
00113                                         string objClassName = obj.getTagName();
00114                                         value = ser.unSerializeUnknown(objxml,arg.getAttribute("type"));
00115                                 }
00116                                 argus.push_back(arg.getAttribute("type"));
00117                                 valus.push_back(value);
00118                         }
00119                         string className = "Component_"+message.getAttribute("beanName");
00120                         _cmp_instance->logger << "\nBean class = " << className << "\n" << flush;
00121                         string returnType = message.getAttribute("returnType");
00122                         string lang = message.getAttribute("lang");
00123                         ClassInfo clas = reflector.getClassInfo(className);
00124                         string methodName = message.getAttribute("name");
00125                         _cmp_instance->logger << "\nBean service = " << methodName << "\n" << flush;
00126                         if(clas.getClassName()=="")
00127                         {
00128                                 throw new ComponentHandlerException("bean does not exist or is not regsitered\n",retValue);
00129                         }
00130                         Method meth = clas.getMethod(methodName,argus);
00131                         if(meth.getMethodName()=="")
00132                         {
00133                                 throw new ComponentHandlerException("service does not exist for the bean or the bean does not exist or is not regsitered\n\n",retValue);
00134                         }
00135                         else
00136                         {
00137                                 _cmp_instance->logger << "\nGot Bean service " << methodName << "\n" << flush;
00138                                 args argus;
00139                                 Constructor ctor = clas.getConstructor(argus);
00140                                 void *_temp = reflector.newInstanceGVP(ctor);
00141                                 _cmp_instance->logger << "\nGot Bean " << _temp << "\n" << flush;
00142                                 if(returnType=="void" || returnType=="")
00143                                 {
00144                                         _cmp_instance->logger << "\nVoid return " << "\n" << flush;
00145                                         reflector.invokeMethod<void*>(_temp,meth,valus);
00146                                         retValue = ("<return:void></return:void>");
00147                                 }
00148                                 else
00149                                 {
00150                                         _cmp_instance->logger << "\nReturn type = "<< returnType << "\n" << flush;
00151                                         if(returnType=="int")
00152                                         {
00153                                                 int retv = reflector.invokeMethod<int>(_temp,meth,valus);
00154                                                 retValue = ("<return:int>"+CastUtil::lexical_cast<string>(retv)+"</return:int>");
00155                                         }
00156                                         else if(returnType=="float")
00157                                         {
00158                                                 float retv = reflector.invokeMethod<float>(_temp,meth,valus);
00159                                                 retValue = ("<return:float>"+CastUtil::lexical_cast<string>(retv)+"</return:float>");
00160                                         }
00161                                         else if(returnType=="double")
00162                                         {
00163                                                 double retv = reflector.invokeMethod<double>(_temp,meth,valus);
00164                                                 retValue = ("<return:double>"+CastUtil::lexical_cast<string>(retv)+"</return:double>");
00165                                         }
00166                                         else if(returnType=="string")
00167                                         {
00168                                                 string retv = reflector.invokeMethod<string>(_temp,meth,valus);
00169                                                 retValue = ("<return:string>"+retv+"</return:string>");
00170                                         }
00171                                         else if(returnType!="")
00172                                         {
00173                                                 void* retobj = reflector.invokeMethodUnknownReturn(_temp,meth,valus);
00174                                                 string oxml = ser.serializeUnknown(retobj,returnType);
00175                                                 retValue = ("<return:"+returnType+">"+oxml+"</return:"+returnType+">");
00176                                         }
00177                                 }
00178 
00179                         }
00180                 }
00181                 else
00182                 {
00183                         retValue = "<return:exception>This is a C++ daemon</return:exception>";
00184                 }
00185                 _cmp_instance->logger << "\nSending data = "<< retValue << "\n" << flush;
00186                 if(retValue!="")
00187                         _cmp_instance->getServer()->Send(fd,retValue);
00188                 //close(fd);
00189         }
00190         catch(ComponentHandlerException *e)
00191         {
00192                 _cmp_instance->logger << e->getMessage() << flush;
00193                 _cmp_instance->getServer()->Send(fd,retValue);
00194                 close(fd);
00195         }
00196         catch(...)
00197         {
00198                 _cmp_instance->logger << "exception occurred" << flush;
00199                 retValue = ("<return:exception>XmlParseException occurred</return:exception>");
00200                 _cmp_instance->getServer()->Send(fd,retValue);
00201                 close(fd);
00202         }
00203 }
