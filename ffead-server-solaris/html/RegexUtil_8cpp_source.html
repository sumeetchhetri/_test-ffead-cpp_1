00036 vector<string> RegexUtil::search(const string& text, const string& pattern) {
00037         vector<string> vec;
00038         string ttext(text);
00039         regex_t regex;
00040         int reti;
00041         if(patterns.find(pattern)!=patterns.end())
00042         {
00043                 regex = patterns[pattern];
00044         }
00045         else
00046         {
00047                 /* Compile regular expression */
00048                 reti = regcomp(&regex, pattern.c_str(), REG_EXTENDED);
00049                 if(reti)
00050                 {
00051                         cout << "Could not compile regex\n" << endl;
00052                 }
00053                 else
00054                 {
00055                         patterns[pattern] = regex;
00056                 }
00057         }
00058         regmatch_t pm;
00059         reti = regexec(&regex, ttext.c_str(), 1, &pm, 0);
00060         while (reti == 0) {    /* while matches found */
00061                 /* substring found between pm.rm_so and pm.rm_eo */
00062                 /* This call to regexec() finds the next match */
00063                 if(!reti) {
00064                         string match;
00065                         match = ttext.substr(pm.rm_so, pm.rm_eo-pm.rm_so);
00066                         vec.push_back(match);
00067                 } else {
00068                         break;
00069                 }
00070                 ttext = ttext.substr(pm.rm_eo);
00071                 pm.rm_eo = -1;
00072                 pm.rm_so = -1;
00073                 reti = regexec (&regex, ttext.c_str(), 1, &pm, 0);
00074         }
00075         return vec;
00076 }
