00034 string FormHandler::handle(HttpRequest* req, HttpResponse& res, map<string, Element> formMap, void* dlib)
00035 {
00036         Logger logger = Logger::getLogger("FormHandler");
00037         Reflector ref;
00038         Element ele = formMap[req->getFile()];
00039         logger << ele.getTagName() << endl;
00040         logger << ele.render() << endl;
00041         ClassInfo binfo = ref.getClassInfo(ele.getAttribute("bean"));
00042         ElementList eles = ele.getChildElements();
00043         string json = "{";
00044         for (unsigned int apps = 0; apps < eles.size(); apps++)
00045         {
00046                 if(eles.at(apps).getTagName()=="field")
00047                 {
00048                         string name = eles.at(apps).getAttribute("name");
00049                         Field fld = binfo.getField(eles.at(apps).getAttribute("prop"));
00050                         if(fld.getType()=="string")
00051                                 json += "\""+eles.at(apps).getAttribute("prop")+"\": \"" + req->getQueryParam(name) + "\",";
00052                         else
00053                         {
00054                                 if(fld.getType()=="int" || fld.getType()=="long")
00055                                 {
00056                                         if(req->getQueryParam(name)=="")
00057                                                 json += "\""+eles.at(apps).getAttribute("prop")+"\": 0,";
00058                                         else
00059                                                 json += "\""+eles.at(apps).getAttribute("prop")+"\": " + req->getQueryParam(name) + ",";
00060                                 }
00061                                 else if(fld.getType()=="double" || fld.getType()=="float")
00062                                 {
00063                                         if(req->getQueryParam(name)=="")
00064                                                 json += "\""+eles.at(apps).getAttribute("prop")+"\": 0.0,";
00065                                         else
00066                                                 json += "\""+eles.at(apps).getAttribute("prop")+"\": " + req->getQueryParam(name) + ",";
00067                                 }
00068                                 else if(fld.getType()=="bool")
00069                                 {
00070                                         if(req->getQueryParam(name)=="")
00071                                                 json += "\""+eles.at(apps).getAttribute("prop")+"\": false,";
00072                                         else
00073                                                 json += "\""+eles.at(apps).getAttribute("prop")+"\": " + req->getQueryParam(name) + ",";
00074                                 }
00075                         }
00076                 }
00077         }
00078         if(json.find(",")!=string::npos)
00079         {
00080                 json = json.substr(0,json.length()-1);
00081         }
00082         json += "}";
00083         logger << json << endl;
00084         string libName = Constants::INTER_LIB_FILE;
00085         if(dlib == NULL)
00086         {
00087                 cerr << dlerror() << endl;
00088                 exit(-1);
00089         }
00090         string meth = "toVoidP" + ele.getAttribute("bean");
00091         logger << meth << endl;
00092         void *mkr = dlsym(dlib, meth.c_str());
00093         if(mkr!=NULL)
00094         {
00095                 toVoidP f1 = (toVoidP)mkr;
00096                 void *_beaninst = f1(json);
00097                 //FunPtr f =  (FunPtr)mkr;
00098                 ClassInfo srv = ref.getClassInfo(ele.getAttribute("controller"));
00099                 args argus;
00100                 vals valus;
00101                 Constructor ctor = srv.getConstructor(argus);
00102                 void *_temp = ref.newInstanceGVP(ctor);
00103                 argus.push_back("void*");
00104                 argus.push_back("HttpResponse*");
00105                 valus.push_back(_beaninst);
00106                 valus.push_back(&res);
00107                 Method meth = srv.getMethod("onSubmit",argus);
00108                 if(meth.getMethodName()!="")
00109                 {
00110                         ref.invokeMethodUnknownReturn(_temp,meth,valus);
00111                         logger << "successfully called formcontroller" << endl;
00112                 }
00113                 else
00114                 {
00115                         res.setStatusCode("404");
00116                         res.setStatusMsg("Not Found");
00117                         res.setContent_type("text/plain");
00118                         res.setContent_str("Controller Method Not Found");
00119                         logger << "Controller Method Not Found" << endl;
00120                 }
00121         }
00122         return json;
00123 }
