00043     template< typename Char_type, class String_type >
00044     bool add_esc_char( Char_type c, String_type& s )
00045     {
00046         switch( c )
00047         {
00048             case '"':  s += to_str< String_type >( "\\\"" ); return true;
00049             case '\\': s += to_str< String_type >( "\\\\" ); return true;
00050             case '\b': s += to_str< String_type >( "\\b"  ); return true;
00051             case '\f': s += to_str< String_type >( "\\f"  ); return true;
00052             case '\n': s += to_str< String_type >( "\\n"  ); return true;
00053             case '\r': s += to_str< String_type >( "\\r"  ); return true;
00054             case '\t': s += to_str< String_type >( "\\t"  ); return true;
00055         }
00056 
00057         return false;
00058     }

00060     template< class String_type >
00061     String_type add_esc_chars( const String_type& s )
00062     {
00063         typedef typename String_type::const_iterator Iter_type;
00064         typedef typename String_type::value_type     Char_type;
00065 
00066         String_type result;
00067 
00068         const Iter_type end( s.end() );
00069 
00070         for( Iter_type i = s.begin(); i != end; ++i )
00071         {
00072             const Char_type c( *i );
00073 
00074             if( add_esc_char( c, result ) ) continue;
00075 
00076             const wint_t unsigned_c( ( c >= 0 ) ? c : 256 + c );
00077 
00078             if( iswprint( unsigned_c ) )
00079             {
00080                 result += c;
00081             }
00082             else
00083             {
00084                 result += non_printable_to_string< String_type >( unsigned_c );
00085             }
00086         }
00087 
00088         return result;
00089     }

00094     template< class Value_type, class Ostream_type >
00095     class Generator
00096     {
00097         typedef typename Value_type::Config_type Config_type;
00098         typedef typename Config_type::String_type String_type;
00099         typedef typename Config_type::Object_type Object_type;
00100         typedef typename Config_type::Array_type Array_type;
00101         typedef typename String_type::value_type Char_type;
00102         typedef typename Object_type::value_type Obj_member_type;
00103 
00104     public:
00105 
00106         Generator( const Value_type& value, Ostream_type& os, bool pretty )
00107         :   os_( os )
00108         ,   indentation_level_( 0 )
00109         ,   pretty_( pretty )
00110         {
00111             output( value );
00112         }
00113 
00114     private:
00115 
00116         void output( const Value_type& value )
00117         {
00118             switch( value.type() )
00119             {
00120                 case obj_type:   output( value.get_obj() );   break;
00121                 case array_type: output( value.get_array() ); break;
00122                 case str_type:   output( value.get_str() );   break;
00123                 case bool_type:  output( value.get_bool() );  break;
00124                 case int_type:   output_int( value );         break;
00125                 case real_type:  os_ << showpoint << setprecision( 16 ) 
00126                                      << value.get_real();     break;
00127                 case null_type:  os_ << "null";               break;
00128                 default: assert( false );
00129             }
00130         }
00131 
00132         void output( const Object_type& obj )
00133         {
00134             output_array_or_obj( obj, '{', '}' );
00135         }
00136 
00137         void output( const Array_type& arr )
00138         {
00139             output_array_or_obj( arr, '[', ']' );
00140         }
00141 
00142         void output( const Obj_member_type& member )
00143         {
00144             output( Config_type::get_name( member ) ); space(); 
00145             os_ << ':'; space(); 
00146             output( Config_type::get_value( member ) );
00147         }
00148 
00149         void output_int( const Value_type& value )
00150         {
00151             if( value.is_uint64() )
00152             {
00153                 os_ << value.get_uint64();
00154             }
00155             else
00156             {
00157                os_ << value.get_int64();
00158             }
00159         }
00160 
00161         void output( const String_type& s )
00162         {
00163             os_ << '"' << add_esc_chars( s ) << '"';
00164         }
00165 
00166         void output( bool b )
00167         {
00168             os_ << to_str< String_type >( b ? "true" : "false" );
00169         }

00171         template< class T >
00172         void output_array_or_obj( const T& t, Char_type start_char, Char_type end_char )
00173         {
00174             os_ << start_char; new_line();
00175 
00176             ++indentation_level_;
00177             
00178             for( typename T::const_iterator i = t.begin(); i != t.end(); ++i )
00179             {
00180                 indent(); output( *i );
00181 
00182                 typename T::const_iterator next = i;
00183 
00184                 if( ++next != t.end())
00185                 {
00186                     os_ << ',';
00187                 }
00188 
00189                 new_line();
00190             }
00191 
00192             --indentation_level_;
00193 
00194             indent(); os_ << end_char;
00195         }
00196         
00197         void indent()
00198         {
00199             if( !pretty_ ) return;
00200 
00201             for( int i = 0; i < indentation_level_; ++i )
00202             { 
00203                 os_ << "    ";
00204             }
00205         }
00206 
00207         void space()
00208         {
00209             if( pretty_ ) os_ << ' ';
00210         }
00211 
00212         void new_line()
00213         {
00214             if( pretty_ ) os_ << '\n';
00215         }
00216 
00217         Ostream_type& os_;
00218         int indentation_level_;
00219         bool pretty_;
00220     };
