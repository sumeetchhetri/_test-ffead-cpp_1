00111     template< class String_type >
00112     void append_esc_char_and_incr_iter( String_type& s, 
00113                                         typename String_type::const_iterator& begin, 
00114                                         typename String_type::const_iterator end )
00115     {
00116         typedef typename String_type::value_type Char_type;
00117              
00118         const Char_type c2( *begin );
00119 
00120         switch( c2 )
00121         {
00122             case 't':  s += '\t'; break;
00123             case 'b':  s += '\b'; break;
00124             case 'f':  s += '\f'; break;
00125             case 'n':  s += '\n'; break;
00126             case 'r':  s += '\r'; break;
00127             case '\\': s += '\\'; break;
00128             case '/':  s += '/';  break;
00129             case '"':  s += '"';  break;
00130             case 'x':  
00131             {
00132                 if( end - begin >= 3 )  //  expecting "xHH..."
00133                 {
00134                     s += hex_str_to_char< Char_type >( begin );  
00135                 }
00136                 break;
00137             }
00138             case 'u':  
00139             {
00140                 if( end - begin >= 5 )  //  expecting "uHHHH..."
00141                 {
00142                     s += unicode_str_to_char< Char_type >( begin );  
00143                 }
00144                 break;
00145             }
00146         }
00147     }

00149     template< class String_type >
00150     String_type substitute_esc_chars( typename String_type::const_iterator begin, 
00151                                    typename String_type::const_iterator end )
00152     {
00153         typedef typename String_type::const_iterator Iter_type;
00154 
00155         if( end - begin < 2 ) return String_type( begin, end );
00156 
00157         String_type result;
00158         
00159         result.reserve( end - begin );
00160 
00161         const Iter_type end_minus_1( end - 1 );
00162 
00163         Iter_type substr_start = begin;
00164         Iter_type i = begin;
00165 
00166         for( ; i < end_minus_1; ++i )
00167         {
00168             if( *i == '\\' )
00169             {
00170                 result.append( substr_start, i );
00171 
00172                 ++i;  // skip the '\'
00173              
00174                 append_esc_char_and_incr_iter( result, i, end );
00175 
00176                 substr_start = i + 1;
00177             }
00178         }
00179 
00180         result.append( substr_start, end );
00181 
00182         return result;
00183     }

00326         template< class Array_or_obj >
00327         void begin_compound()
00328         {
00329             if( current_p_ == 0 )
00330             {
00331                 add_first( Array_or_obj() );
00332             }
00333             else
00334             {
00335                 stack_.push_back( current_p_ );
00336 
00337                 Array_or_obj new_array_or_obj;   // avoid copy by building new array or object in place
00338 
00339                 current_p_ = add_to_current( new_array_or_obj );
00340             }
00341         }
00342 
00343         void end_compound()
00344         {
00345             if( current_p_ != &value_ )
00346             {
00347                 current_p_ = stack_.back();
00348                 
00349                 stack_.pop_back();
00350             }

00354         {
00355             if( current_p_ == 0 )
00356             {
00357                 return add_first( value );
00358             }
00359             else if( current_p_->type() == array_type )
00360             {
00361                 current_p_->get_array().push_back( value );
00362 
00363                 return &current_p_->get_array().back(); 
00364             }
00365             
00366             assert( current_p_->type() == obj_type );
00367 
00368             return &Config_type::add( current_p_->get_obj(), name_, value );
00369         }

00371         Value_type& value_;             // this is the object or array that is being created
00372         Value_type* current_p_;         // the child object or array that is currently being constructed

00374         vector< Value_type* > stack_;   // previous child objects and arrays
00375 
00376         String_type name_;              // of current name/value pair
00377     };

00435         template< typename ScannerT >
00436         struct definition
00437         {
00438             definition( const Json_grammer& self )
00439             {
00440                 typedef typename Value_type::String_type::value_type Char_type;
00441 
00442                 // first we convert the semantic action class methods to functors with the 
00443                 // parameter signature expected by spirit
00444 
00445                 typedef function< void( Char_type )            > Char_action;
00446                 typedef function< void( Iter_type, Iter_type ) > Str_action;
00447                 typedef function< void( double )               > Real_action;
00448                 typedef function< void( int64_t )              > Int_action;
00449                 typedef function< void( uint64_t )             > Uint64_action;
00450 
00451                 Char_action   begin_obj  ( bind( &Semantic_actions_t::begin_obj,   &self.actions_, _1 ) );
00452                 Char_action   end_obj    ( bind( &Semantic_actions_t::end_obj,     &self.actions_, _1 ) );
00453                 Char_action   begin_array( bind( &Semantic_actions_t::begin_array, &self.actions_, _1 ) );
00454                 Char_action   end_array  ( bind( &Semantic_actions_t::end_array,   &self.actions_, _1 ) );
00455                 Str_action    new_name   ( bind( &Semantic_actions_t::new_name,    &self.actions_, _1, _2 ) );
00456                 Str_action    new_str    ( bind( &Semantic_actions_t::new_str,     &self.actions_, _1, _2 ) );
00457                 Str_action    new_true   ( bind( &Semantic_actions_t::new_true,    &self.actions_, _1, _2 ) );
00458                 Str_action    new_false  ( bind( &Semantic_actions_t::new_false,   &self.actions_, _1, _2 ) );
00459                 Str_action    new_null   ( bind( &Semantic_actions_t::new_null,    &self.actions_, _1, _2 ) );
00460                 Real_action   new_real   ( bind( &Semantic_actions_t::new_real,    &self.actions_, _1 ) );
00461                 Int_action    new_int    ( bind( &Semantic_actions_t::new_int,     &self.actions_, _1 ) );
00462                 Uint64_action new_uint64 ( bind( &Semantic_actions_t::new_uint64,  &self.actions_, _1 ) );
00463 
00464                 // actual grammer
00465 
00466                 json_
00467                     = value_ | eps_p[ &throw_not_value ]
00468                     ;
00469 
00470                 value_
00471                     = string_[ new_str ] 
00472                     | number_ 
00473                     | object_ 
00474                     | array_ 
00475                     | str_p( "true" ) [ new_true  ] 
00476                     | str_p( "false" )[ new_false ] 
00477                     | str_p( "null" ) [ new_null  ]
00478                     ;
00479 
00480                 object_ 
00481                     = ch_p('{')[ begin_obj ]
00482                     >> !members_
00483                     >> ( ch_p('}')[ end_obj ] | eps_p[ &throw_not_object ] )
00484                     ;
00485 
00486                 members_

00488                     ;
00489 
00490                 pair_
00491                     = string_[ new_name ]
00492                     >> ( ':' | eps_p[ &throw_not_colon ] )
00493                     >> ( value_ | eps_p[ &throw_not_value ] )

00496                 array_
00497                     = ch_p('[')[ begin_array ]
00498                     >> !elements_
00499                     >> ( ch_p(']')[ end_array ] | eps_p[ &throw_not_array ] )
00500                     ;

00502                 elements_
00503                     = value_ >> *( ',' >> value_ )
00504                     ;
00505 
00506                 string_
