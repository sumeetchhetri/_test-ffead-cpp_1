00103         {
00104                 map<K,V> mpt  = mp;
00105                 string objXml;
00106                 objXml = "<map>";
00107                 while (mpt.begin()!=mpt.end())
00108                 {
00109                         objXml += "<pair>";
00110                         objXml += serialize<K>(mpt.begin()->first);
00111                         objXml += serialize<V>(mpt.begin()->second);
00112                         mpt.erase(mpt.begin());
00113                         objXml += "</pair>";
00114                 }
00115                 objXml += "</map>";
