00307     template< class Config >
00308     Value_impl< Config >::Value_impl( const Value_impl< Config >& other )
00309     :   type_( other.type() )
00310     ,   is_uint64_( other.is_uint64_ )
00311     {
00312         switch( type_ )
00313         {
00314             case str_type:   str_     = other.get_str();                               break;
00315             case obj_type:   obj_p_   = Object_ptr( new Object( other.get_obj() ) );   break;
00316             case array_type: array_p_ = Array_ptr ( new Array ( other.get_array() ) ); break;
00317             case bool_type:  bool_    = other.get_bool();                              break;
00318             case int_type:   i_       = other.get_int64();                             break;
00319             case real_type:  d_       = other.get_real();                              break;
00320             case null_type:                                                            break;
00321             default: assert( false );
00322         };
00323     }

00327     {

00347     template< class Config >
00348     bool Value_impl< Config >::operator==( const Value_impl& lhs ) const
00349     {
00350         if( this == &lhs ) return true;
00351 
00352         if( type() != lhs.type() ) return false;
00353 
00354         switch( type_ )
00355         {
00356             case str_type:   return get_str()   == lhs.get_str();
00357             case obj_type:   return get_obj()   == lhs.get_obj();
00358             case array_type: return get_array() == lhs.get_array();
00359             case bool_type:  return get_bool()  == lhs.get_bool();
00360             case int_type:   return ( get_int64() == lhs.get_int64() ) &&
00361                                     ( is_uint64() == lhs.is_uint64() );
00362             case real_type:  return get_real()  == lhs.get_real();
00363             case null_type:  return true;
00364         };
00365 
00366         assert( false );
00367 
00368         return false; 
00369     }
