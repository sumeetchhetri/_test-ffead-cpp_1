00030 Message::Message(string xml)
00031 {
00032         try
00033         {
00034                 XmlParser parser("Parser");
00035                 Document doc = parser.getDocument(xml);
00036                 Element message = doc.getRootElement();
00037                 if(message.getTagName()!="message")
00038                 {
00039                         throw new InvalidMessageException("No message Tag\n");
00040                 }
00041                 else if(message.getChildElements().size()!=2)
00042                 {
00043                         throw new InvalidMessageException("Every message should have a headers and body tag\n");
00044                 }
00045                 Element headers = message.getChildElements().at(0);
00046                 Element body = message.getChildElements().at(1);
00047                 if(headers.getTagName()!="headers")
00048                 {
00049                         throw new InvalidMessageException("No headers Tag\n");
00050                 }
00051                 else if(body.getTagName()!="body")
00052                 {
00053                         throw new InvalidMessageException("No body Tag\n");
00054                 }
00055                 Element destination = headers.getElementByName("destination");
00056                 Element encoding = headers.getElementByName("encoding");
00057                 Element timestamp = headers.getElementByName("timestamp");
00058                 Element priority = headers.getElementByName("priority");
00059                 Element type = headers.getElementByName("type");
00060                 Element userid = headers.getElementByName("userid");
00061                 if(destination.getTagName()!="destination")
00062                 {
00063                         throw new InvalidMessageException("Destination Header is mandatory\n");
00064                 }
00065                 else if (destination.getAttributes().size()!=2)
00066                 {
00067                         throw new InvalidMessageException("Type and Name Attributes should be speciifes for a Destination\n");
00068                 }
00069                 else if (destination.getAttribute("name")=="" || destination.getAttribute("type")=="")
00070                 {
00071                         throw new InvalidMessageException("Type and Name Attributes cannot be blank for a Destination\n");
00072                 }
00073                 else if(type.getTagName()!="type")
00074                 {
00075                         throw new InvalidMessageException("Type Header is mandatory\n");
00076                 }
00077                 Destination des;
00078                 des.setName(destination.getAttribute("name"));
00079                 des.setType(destination.getAttribute("type"));
00080                 this->destination = des;
00081                 this->body = body.getText();
00082                 this->timestamp = timestamp.getText();
00083                 this->type = type.getText();
00084                 this->priority = priority.getText();
00085                 this->userId = userid.getText();
00086                 this->encoding = encoding.getText();
00087         }
