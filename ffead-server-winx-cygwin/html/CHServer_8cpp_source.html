00081         msg.msg_iovlen = 1;

00085         msg.msg_control = ccmsg;

00088         cmsg->cmsg_level = SOL_SOCKET;
00089         cmsg->cmsg_type = SCM_RIGHTS;
00090         cmsg->cmsg_len = CMSG_LEN(sizeof(int));
00091         *(int*)CMSG_DATA(cmsg) = fd;
00092         msg.msg_controllen = cmsg->cmsg_len;
00093 
00094         msg.msg_flags = 0;
00095         //boost::mutex::scoped_lock lock(p_mutex);
00096         if((rv= sendmsg(unix_socket_fd, &msg, 0)) < 0 )
00097         {
00098           perror("sendmsg()");

00100         }
00101         close(fd);
00102         return rv;
